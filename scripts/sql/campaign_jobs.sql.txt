-- Campaign Jobs minimal schema (idempotent)
BEGIN;

CREATE TABLE IF NOT EXISTS campaign_jobs (
  id TEXT PRIMARY KEY,
  doctor_id TEXT NOT NULL,
  campaign_id TEXT NOT NULL,
  channel TEXT NOT NULL CHECK (channel in ('whatsapp','sms','email')),
  trigger TEXT NULL,
  schedule_at TIMESTAMPTZ NOT NULL,
  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  status TEXT NOT NULL CHECK (status in ('scheduled','running','done','failed','cancelled')),
  last_error TEXT NULL
);

CREATE INDEX IF NOT EXISTS idx_campaign_jobs_doctor_id ON campaign_jobs(doctor_id);
CREATE INDEX IF NOT EXISTS idx_campaign_jobs_schedule_at ON campaign_jobs(schedule_at);
CREATE INDEX IF NOT EXISTS idx_campaign_jobs_status ON campaign_jobs(status);

COMMIT;
